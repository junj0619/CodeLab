## 621. Task Scheduler

```java

class Solution {
    public int leastInterval(char[] tasks, int n) {
        int len = tasks.length;
        int[] taskCT = new int[26];
        for (char task : tasks) {
            ++taskCT[task - 'A'];
        }
        
        Arrays.sort(taskCT);
        
        int max = taskCT[25], res = 0, i = 25;
        
        while (i >= 0 && taskCT[i] == max) i--;                
        res = (max - 1) * (n + 1) + (25 - i);
        
        return Math.max(res, len);
    }
}
```
