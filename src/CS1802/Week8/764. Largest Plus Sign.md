## 764. Largest Plus Sign ##


### Brute Force(N^3) ###
```java
class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int ans = 0;
        int[][] matrix = new int[N][N];
        
        for (int[] m : matrix) {
            Arrays.fill(m, 1);
        }
        
        for (int[] mine : mines) {
            int x = mine[0], y = mine[1];
            matrix[x][y] = 0;
        }
        
        for (int r = 0; r < N; r++) {
            for (int c = 0; c < N; c++) {
                if (matrix[r][c] == 0) continue;
                
                int lvl = 1;
                int order = 1;
                
                //int up = r - lvl, down = r + lvl, left = c - lvl, right = c + lvl;
                while (r - lvl >= 0 && r + lvl < N && c - lvl >= 0 && c + lvl < N 
                       && matrix[r - lvl][c] == 1 && matrix[r + lvl][c] == 1 
                       && matrix[r][c - lvl] == 1 && matrix[r][c + lvl] == 1) {
                    lvl++;
                    order++;
                }           
                
                ans = Math.max(ans, order);
            }
        }
        
        return ans;
    }
}
```

### DP ###

```java

class Solution {
    public int orderOfLargestPlusSign(int N, int[][] mines) {
        int ans = 0;
        int[][] dp = new int[N][N];
        
        for (int[] d : dp) {
            Arrays.fill(d, 1);
        }
        
        for (int[] mine : mines) {
            dp[mine[0]][mine[1]] = 0;
        }
        //from left
        for (int r = 0; r < N; r++) {
            int count = 0;
            for (int c = 0; c < N; c++) {
                count = dp[r][c] == 0 ? 0 : count + 1;
                dp[r][c] = count;
            }            
        }
        //from right
        for (int r = N - 1; r >= 0; r--) {
            int count = 0;
            for (int c = N - 1; c >= 0; c--) {
                count = dp[r][c] == 0 ? 0 : count + 1;
                dp[r][c] = Math.min(dp[r][c], count);
            }
        }
        //from top
        for (int c = 0; c < N; c++) {
            int count = 0;
            for (int r = 0; r < N; r++) {
                count = dp[r][c] == 0 ? 0 : count + 1;
                dp[r][c] = Math.min(dp[r][c], count);            
            }
        }
        //from bottom
        for (int c = 0; c < N; c++) {
            int count = 0;
            for (int r = N - 1; r >= 0; r--) {
                count = dp[r][c] == 0 ? 0 : count + 1;
                dp[r][c] = Math.min(dp[r][c], count);
                ans = Math.max(ans, dp[r][c]);
            }
        }
        
        return ans;        
    }
}
```
