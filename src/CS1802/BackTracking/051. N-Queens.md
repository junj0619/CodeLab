## 51. N-Queens

```java
class Solution {
    List<List<String>> res;
    public List<List<String>> solveNQueens(int n) {        
        res = new ArrayList<>();        
        char[][] chess = new char[n][n];
        
        for (int i = 0; i < n; i++) {
            Arrays.fill(chess[i], '.');
        }
        
        solve(0, chess);
        
        return res;
    }
    
    private void solve(int row, char[][] chess) {
        if (row == chess.length) {
            res.add(build(chess));
            return;
        }
        
        for (int col = 0; col < chess.length; col++) {
            if (!isValid(chess, row, col)) continue;
            
            chess[row][col] = 'Q';            
            solve(row + 1, chess);            
            chess[row][col] = '.';
        }
    }
    
    private boolean isValid(char[][] chess, int row, int col) {
        
        for (int r = row - 1; r >= 0; r--) {
            if (chess[r][col] == 'Q') 
                return false;
        }
        
        for (int r = row - 1, c = col - 1; r >= 0 && c >= 0; r--, c--) {
            if (chess[r][c] == 'Q')
                return false;
        }
        
        for (int r = row - 1, c = col + 1; r >= 0 && c < chess.length; r--, c++) {
            if (chess[r][c] == 'Q')
                return false;
        }
        
        return true;
        
    }
    
    private List<String> build(char[][] chess) {
        List<String> list = new ArrayList<>();
        for (int row = 0; row < chess.length; row++) {
            list.add(new String(chess[row]));
        }
        return list;
    }
}
```
