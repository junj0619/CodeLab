## 839. Similar String Groups ##

class Solution {
    public int numSimilarGroups(String[] A) {        
        int len = A.length;
        int wordLen = A[0].length();
        
        HashMap<String, Integer> map = new HashMap<>(); // Key: String, Value: Index
        int[] un = new int[len];
        
        for (int i = 0; i < len; i++) {
            map.put(A[i], i);
            un[i] = i;
        }
        
        for (int j = 0; j < len; j++) {                   
            for (int s = 0; s < wordLen - 1; s++) {
                for (int e = s + 1; e < wordLen; e++) {
                    String str = swap(A[j], s, e);
                    if (!map.containsKey(str)) continue;
                    union(un, j, map.get(str));
                }
            }            
        }
        
        Set<Integer> set = new HashSet<>();
        for (int k = 0; k < len; k++) {
            int u = find(un, k);
            if (!set.contains(u))
                set.add(u);
        }
        
        return set.size();        
    }
    
    private String swap(String str, int p, int q) {
        char[] chars = str.toCharArray();
        char temp = chars[p];
        chars[p] = chars[q];
        chars[q] = temp;
        return new String(chars);
    }
    
    private void union(int[] un, int a, int b){
        int parentA = find(un, a);
        int parentB = find(un, b);
        
        un[parentB] = parentA;
    }
    
    private int find(int[] un, int el) {
        while (el != un[el]) {
            el = un[el];
        }
        return el;
    }    
    
}

class Solution {
    public int numSimilarGroups(String[] A) {  
        int ans = 0;
        int len = A.length;
        int wordLen = A[0].length();
        
        HashMap<String, Integer> map = new HashMap<>(); // Key: String, Value: Index
        int[] un = new int[len];
        
        for (int i = 0; i < len; i++) {
            map.put(A[i], i);
            un[i] = i;
        }
        
        for (int j = 0; j < len - 1; j++) {
            for (int l = j + 1; l < len; l++) {
                if (isSimilar(A[j], A[l])) {
                    union(un, j, l);
                }
            }
        }
        
     
        for (int k = 0; k < len; k++) {
            int u = find(un, k);
            if (u == k) ans++;
        }
        
        return ans;        
    }
    
    
    private boolean isSimilar(String w1, String w2){
        int diff = 0;
        for (int n = 0; n < w1.length(); n++) {
            if (w1.charAt(n) != w2.charAt(n))
                diff++;
            if (diff > 2)
                break;
        }
        return diff <= 2;
    }
    
    private void union(int[] un, int a, int b){
        int parentA = find(un, a);
        int parentB = find(un, b);
        
        un[parentB] = parentA;
    }
    
    private int find(int[] un, int el) {
        while (el != un[el]) {
            el = un[el];
        }
        return el;
    }    
    
}
