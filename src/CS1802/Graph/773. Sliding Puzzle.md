## 773. Sliding Puzzle


Idea:
1) Convert board to a string
2) Move zero 4 dictrections base on cur board string
3) BFS level traversal count step
4) Return steps if target string is found

```java
class Solution {
    public int slidingPuzzle(int[][] board) {
        int steps = 0;
        int rows = board.length, cols = board[0].length;        
        
        Deque<String> queue = new ArrayDeque<>();
        int[][] dicts = {{-1, 0}, {1, 0}, {0, -1}, {0, 1}};        
        String str = "", target = "123450";
        Set<String> visited = new HashSet<>();
        
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                str += board[i][j];
            }
        }
        
        queue.offer(str);
       
        
        while (!queue.isEmpty()) {    
            
            int size = queue.size();           
            
            for (int i = 0; i < size; i++) {
                 String cur = queue.poll();
                 if (cur.equals(target)) return steps;
            
                 int index = cur.indexOf("0");
                 int row = index / cols;
                 int col = index % cols;
                
                 for (int[] dict : dicts) {
                     int newRow = dict[0] + row;
                     int newCol = dict[1] + col;
                     int newIndex = newRow * cols + newCol;

                     if (newRow < 0 || newRow >= rows || newCol < 0 || newCol >= cols) continue;

                     String next = swap(cur, newIndex, index);
                     if (visited.contains(next)) continue;                                                                        
                     queue.offer(next);
                     visited.add(next);                    
                }  
            }                                             
            steps++;
        }
        
        return -1;        
    }
    
    private String swap (String str, int i, int j) {
        char[] chars = str.toCharArray();
        char temp = chars[i];
        chars[i] = chars[j];
        chars[j] = temp;
        
        return new String(chars);
    }
}
```
